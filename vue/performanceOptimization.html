若要关闭或者修改，需手动配置

// vue.config.js
module.exports = {
  chainWebpack: config => {
    // 移除 prefetch 插件
    config.plugins.delete('prefetch')

    // 或者
    // 修改它的选项：
    config.plugin('prefetch').tap(options => {
      options[0].fileBlacklist = options[0].fileBlacklist || []
      options[0].fileBlacklist.push(/myasyncRoute(.)+?\.js$/)
      return options
    })
  }
}

# 静态资源

  1. url 以'.'开头会被根据文件的相对路径解析
  2. url 以'~'开头会被当作模块解析
  3. url 以'/'不会转换，直接引用public的资源，public里面的文件会被直接复制，不经过webpack编译 public

# 全局样式

// vue.config.js
module.exports = {
  css: {
    loaderOptions: {
      // 给 sass-loader 传递选项
      sass: {
        // @/ 是 src/ 的别名
        // 所以这里假设你有 `src/variables.scss` 这个文件
        data: `@import "~@/variables.scss";`
      }
    }
  }
}

# tslint简单用法

vue装饰器vue-property-decorator

<script lang='ts'>
import { Vue, Component, Prop } from 'vue-property-decorator';
enum Color { Red = 1, Green = 10, Blue }

@Component
export default class About extends Vue {
  @Prop(Number) public readonly propA!: number;

  public list: number[] = [1, 1];
  public color: Color = Color.Blue;

  private mounted() {
    console.log(this.color);
  }

  private goPublic() {
    console.log(this.list);
  }
}
</script>
