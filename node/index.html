# 邮件

  1. 使用nodemailer

注意事项

1.QQ 的 host 是 smtp.qq.com、163 的 host 是 smtp.163.com

2.设置 secureConnection: true, port: 465,secure: true

3.163的可以pass可以直接使用邮箱密码，qq的要使用授权码

4.from: 'cs112751cs112751@163.com',发送地址参照个这个格式写


























 
 
 









 




















//app.js
let express = require('express');
let bodyParser = require('body-parser');
let nodemailer = require('nodemailer')
let fs = require('fs')
let app = express();
app.use(express.static('public'));
app.use(bodyParser.urlencoded({ extended: false }))

// parse application/json
app.use(bodyParser.json())

// 设置跨域
app.all('*', function (req, res, next) {
  res.header("Access-Control-Allow-Credentials", "true")
  res.header("Access-Control-Allow-Origin", "*")
  res.header('Access-Control-Allow-Headers', 'Content-Type')
  res.header("Access-Control-Allow-Methods", "PUT,POST,GET,DELETE,OPTIONS")
  res.header("X-Powered-By", ' 3.2.1')
  res.header("Content-Type", "application/json;charset=utf-8")
  next()
})
async function main() {
  // create reusable transporter object using the default SMTP transport
  let transporter = nodemailer.createTransport({
    secureConnection: true,
    host: 'smtp.163.com',
    port: 465,
    secure: true, // true for 465, false for other ports
    auth: {
      user: 'cs112751@163.com', // generated ethereal user
      pass: 'cs112724' // generated ethereal password
    }
  })

  // send mail with defined transport object
  let info = await transporter.sendMail({
    from: 'cs112751<cs112751@163.com>', // sender address
    to: 'cs112751@163.com', // list of receivers
    subject: 'Hello ✔', // Subject line
    text: 'Hello world?', // plain text body
    html: '<b>Hello world?</b>' // html body
  })

  console.log("Message sent: %s", info.messageId);
  // Message sent: <b658f8ca-6296-ccf4-8306-87d57a0b4321@example.com>

  // Preview only available when sending through an Ethereal account
  // console.log("Preview URL: %s", nodemailer.getTestMessageUrl(info));
  // Preview URL: https://ethereal.email/message/WaQKMgKddxQDoou...
}

main().catch(console.error);



app.listen(3000)
